CREATE TABLE peliculas(ID BIGINT PRIMARY KEY,
 url VARCHAR(255),

 title VARCHAR(255),

 duration BIGINT,

name VARCHAR(255),
 img_url VARCHAR(255));


 CREATE TABLE peliculas_infocast(
peliculas_id BIGINT NOT NULL,
infocast_id BIGINT NOT NULL,
PRIMARY KEY (peliculas_id, infocast_id),
foreign key (peliculas_id) references peliculas(id),
foreign key (infocast_id) references infocast(id)
);
